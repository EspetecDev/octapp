```typescript
// RESEARCH.md Reward Reveal on Party Page pattern (Pitfall 4):
let showRewardReveal = $state(false);
let revealedChapterIndex = $state<number | null>(null);

$effect(() => {
  const idx = $gameState?.activeChapterIndex ?? null;
  const chapter = idx !== null ? $gameState?.chapters[idx] : null;
  if (!initialSyncDone) {
    // Baseline on first STATE_SYNC
    // If scavengerDone is already true, show overlay immediately (player joined mid-reward)
    revealedChapterIndex = idx;
    if (chapter?.scavengerDone && idx !== null) {
      showRewardReveal = true;
    }
    return;
  }
  // New reward: chapter advanced OR scavengerDone just flipped
  if (chapter?.scavengerDone && idx !== revealedChapterIndex) {
    revealedChapterIndex = idx;
    showRewardReveal = true;
  }
  // Dismiss when chapter advances (next UNLOCK_CHAPTER changes idx)
  if (idx !== revealedChapterIndex && showRewardReveal) {
    showRewardReveal = false;
  }
});
```

Note: This interacts with the EXISTING `initialSyncDone` state — the baseline guard already prevents stale recap cards. The reward reveal effect must be a SEPARATE `$effect` to avoid interfering with the recap card logic.
